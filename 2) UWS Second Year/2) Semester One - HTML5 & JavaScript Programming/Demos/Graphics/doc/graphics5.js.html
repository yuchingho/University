var canvas, context, cellNo, img, w, h, x_pos;

function drawCell(index) {
    var x = w * (index % 8);
    context.drawImage(img, x, 0, w, h,
        x_pos, 20, w, h);
    x_pos += 10;
}

function animate() {
    context.clearRect(20, 20,
        x_pos + w, 20 + h);
    drawCell(cellNo);
    cellNo += 1;
    if (cellNo === 8) {
        cellNo = 0;
    }
    //x_pos += 10;
}

window.onload = function () {
    x_pos=20;
    canvas = document.getElementById("c");
    context = canvas.getContext("2d");
    img = new Image();
    img.onload = function () {
        cellNo = 0;
        w = img.width / 8;
        h = img.height;
        setInterval(animate, 40);
    };
    img.src = "sonicstrip.png";
};
